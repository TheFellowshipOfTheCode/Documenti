\subsubsection{Informazioni generali}
\label{QuizziPedia::Front-End::ModelViews}
\begin{figure} [ht]
	\centering
	\includegraphics[scale=0.45]{UML/Package/QuizziPedia_Front-End_ModelViews.png}

\begin{itemize}
	\item \textbf{Descrizione}: package contenente le classi che saranno presenti nella variabile d'ambiente \texttt{\$scope} di \textit{Angular\ped{G}} che permettono il \textit{Two-Way Data-Binding\ped{G}} tra le \textit{views\ped{G}} e i \textit{controllers\ped{G}};
	\item \textbf{Padre}: \texttt{Front-End};
	\item \textbf{Interazione con altri componenti}:
	\begin{itemize}
		\item \texttt{Controllers}: \textit{package\ped{G}} contenente i \textit{controllers\ped{G}} front-end dell'applicazione;
		\item \texttt{Directives}: \textit{package\ped{G}} contenente le \textit{directives\ped{G}} front-end dell'applicazione;
		\item \texttt{View}: \textit{package\ped{G}} contenente le \textit{views\ped{G}} front-end dell'applicazione;
		\item \texttt{Templates}: \textit{package\ped{G}} contenente i templates necessari per la creazione dinamica delle viste per le domande.
	\end{itemize}
\end{itemize}
