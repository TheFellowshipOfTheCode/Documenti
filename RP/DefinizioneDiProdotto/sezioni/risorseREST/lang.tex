			\item \textbf{Livello di Accesso}: utente;

\begin{lstlisting}[language=json,firstnumber=1]
{
"lang" : [ita | eng]
"ita" : [array JSON con associazione: "variabile" : "traduzione"]
"eng" : [array JSON con associazione: "variabile" : "traduzione"]
}
\end{lstlisting}
		\end{itemize}
